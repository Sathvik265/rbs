-- Helper function to get current open shift session for a given shift type and date
CREATE OR REPLACE FUNCTION get_current_shift_session_id(
    shift_type VARCHAR DEFAULT NULL, 
    target_date DATE DEFAULT NULL,
    clerk_initials_param VARCHAR DEFAULT NULL
) RETURNS UUID AS $function_body$
DECLARE
    shift_session_id UUID;
    func_date DATE;
    func_time TIME;
BEGIN
    -- Use provided date or current date
    func_date := COALESCE(target_date, CURRENT_DATE);
    func_time := CURRENT_TIME;
    
    -- If shift_type is not specified, determine based on current time
    IF shift_type IS NULL THEN
        IF func_time BETWEEN TIME '06:00:00' AND TIME '11:59:59' THEN
            shift_type := '`';
        ELSIF func_time BETWEEN TIME '12:00:00' AND TIME '17:59:59' THEN
            shift_type := '``';
        ELSIF func_time BETWEEN TIME '18:00:00' AND TIME '21:59:59' THEN
            shift_type := 'RBS1';
        ELSE
            shift_type := 'RBS2';
        END IF;
    END IF;
    
    -- Get the shift session ID for the specified type and date where status is OPEN
    SELECT ss.shift_session_id INTO shift_session_id
    FROM shift_sessions ss
    WHERE ss.shift_name = shift_type 
      AND ss.session_date = func_date 
      AND ss.status = 'OPEN'
      AND (clerk_initials_param IS NULL OR ss.clerk_initials = clerk_initials_param)
    ORDER BY ss.start_time DESC
    LIMIT 1;
    
    -- If no open shift session found, try to find any shift session for that day/type
    IF shift_session_id IS NULL THEN
        SELECT ss.shift_session_id INTO shift_session_id
        FROM shift_sessions ss
        WHERE ss.shift_name = shift_type 
          AND ss.session_date = func_date
          AND (clerk_initials_param IS NULL OR ss.clerk_initials = clerk_initials_param)
        ORDER BY ss.start_time DESC
        LIMIT 1;
    END IF;
    
    RETURN shift_session_id;
END;
$function_body$ LANGUAGE plpgsql;

-- Function to move orders to bill_items when bill is printed
CREATE OR REPLACE FUNCTION move_orders_to_bill_items(
    p_track VARCHAR(20),
    p_clerk_initials VARCHAR(10),
    p_table_no VARCHAR(20),
    p_party_no VARCHAR(20),
    p_bill_number INTEGER
) RETURNS INTEGER AS $$
DECLARE
    moved_count INTEGER := 0;
    bill_id_val INTEGER;
BEGIN
    -- Get the bill_id for the composite key
    SELECT id INTO bill_id_val
    FROM bills
    WHERE track = p_track
      AND clerk_initials = p_clerk_initials
      AND table_no = p_table_no
      AND party_no = p_party_no
      AND bill_number = p_bill_number;

    -- If bill doesn't exist, return 0
    IF bill_id_val IS NULL THEN
        RETURN 0;
    END IF;

    -- Insert orders into bill_items
    INSERT INTO bill_items (
        bill_id, track, clerk_initials, table_no, party_no, bill_number,
        item_code, numeric_item_code, item_name, quantity, unit_price, line_total
    )
    SELECT 
        bill_id_val, track, clerk_initials, table_no, party_no, bill_number,
        item_code, numeric_item_code, item_name, quantity, unit_price, line_total
    FROM orders
    WHERE track = p_track
      AND clerk_initials = p_clerk_initials
      AND table_no = p_table_no
      AND party_no = p_party_no
      AND bill_number = p_bill_number;

    GET DIAGNOSTICS moved_count = ROW_COUNT;

    -- Delete the orders after moving to bill_items
    DELETE FROM orders
    WHERE track = p_track
      AND clerk_initials = p_clerk_initials
      AND table_no = p_table_no
      AND party_no = p_party_no
      AND bill_number = p_bill_number;

    RETURN moved_count;
END;
$$ LANGUAGE plpgsql;

-- Function to check if there are pending orders for a bill
CREATE OR REPLACE FUNCTION has_pending_orders(
    p_track VARCHAR(20),
    p_clerk_initials VARCHAR(10),
    p_table_no VARCHAR(20),
    p_party_no VARCHAR(20),
    p_bill_number INTEGER
) RETURNS BOOLEAN AS $$
DECLARE
    order_count INTEGER;
BEGIN
    SELECT COUNT(*) INTO order_count
    FROM orders
    WHERE track = p_track
      AND clerk_initials = p_clerk_initials
      AND table_no = p_table_no
      AND party_no = p_party_no
      AND bill_number = p_bill_number;

    RETURN order_count > 0;
END;
$$ LANGUAGE plpgsql;

-- Function to get section name by table number
CREATE OR REPLACE FUNCTION get_section_by_table(p_table_no VARCHAR(20))
RETURNS VARCHAR AS $$
DECLARE
    section_name VARCHAR(100);
    table_num INTEGER;
BEGIN
    -- Convert table_no to integer for lookup
    BEGIN
        table_num := p_table_no::INTEGER;
    EXCEPTION WHEN OTHERS THEN
        RETURN 'Unknown';
    END;

    SELECT s.section_name INTO section_name
    FROM sections s
    WHERE s.table_id = table_num;

    RETURN COALESCE(section_name, 'Unknown');
END;
$$ LANGUAGE plpgsql;

-- View for pending bills (bills with orders not yet printed)
CREATE OR REPLACE VIEW pending_bills AS
SELECT DISTINCT
    o.track,
    o.clerk_initials,
    o.table_no,
    o.party_no,
    o.bill_number,
    get_section_by_table(o.table_no) as section_name,
    COUNT(o.id) as total_items,
    SUM(o.line_total) as total_amount,
    MIN(o.created_at) as order_started_at,
    MAX(o.updated_at) as last_updated_at
FROM orders o
GROUP BY o.track, o.clerk_initials, o.table_no, o.party_no, o.bill_number;

-- View for completed bills (bills that have been printed)
CREATE OR REPLACE VIEW completed_bills AS
SELECT DISTINCT
    bi.track,
    bi.clerk_initials,
    bi.table_no,
    bi.party_no,
    bi.bill_number,
    get_section_by_table(bi.table_no) as section_name,
    COUNT(bi.id) as total_items,
    SUM(bi.line_total) as items_total,
    b.grand_total,
    MIN(bi.created_at) as bill_printed_at
FROM bill_items bi
LEFT JOIN bills b ON (
    bi.track = b.track AND 
    bi.clerk_initials = b.clerk_initials AND 
    bi.table_no = b.table_no AND 
    bi.party_no = b.party_no AND 
    bi.bill_number = b.bill_number
)
GROUP BY bi.track, bi.clerk_initials, bi.table_no, bi.party_no, bi.bill_number, b.grand_total;

-- View for table status
CREATE OR REPLACE VIEW table_status AS
SELECT 
    s.table_id,
    s.section_name,
    CASE 
        WHEN pb.table_no IS NOT NULL THEN 'OCCUPIED'
        ELSE 'AVAILABLE'
    END as status,
    pb.total_items,
    pb.total_amount,
    pb.order_started_at
FROM sections s
LEFT JOIN pending_bills pb ON s.table_id::VARCHAR = pb.table_no
ORDER BY s.table_id;
