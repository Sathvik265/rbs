-- Restaurant Billing System Database Schema - Updated Version
-- PostgreSQL 17.6 compatibility
-- Merged sessions and shifts into shift_sessions table
-- Removed specified columns from bills table

DROP TABLE IF EXISTS 
    settings, 
    items, 
    shift_sessions,  -- New merged table
    bill_sequences, 
    bills, 
    bill_items, 
    audit_log 
CASCADE;

-- Enable UUID extension if not exists
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- 1. Settings table for system configuration (unchanged)
CREATE TABLE IF NOT EXISTS settings (
    id SERIAL PRIMARY KEY,
    hotel_name VARCHAR(255) DEFAULT 'Restaurant Name',
    address TEXT,
    phone VARCHAR(20),
    gstin VARCHAR(20),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 2. Items/Menu table (unchanged)
CREATE TABLE IF NOT EXISTS items (
    id SERIAL PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    alpha_code VARCHAR(20) UNIQUE,
    numeric_code VARCHAR(20) UNIQUE,
    price_fixed DECIMAL(10,2) DEFAULT 0,
    price_general DECIMAL(10,2) DEFAULT 0,
    price_ac DECIMAL(10,2) DEFAULT 0,
    category VARCHAR(100),
    item_group VARCHAR(50) DEFAULT 'regular',
    is_active BOOLEAN DEFAULT true,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 3. NEW: Merged shift_sessions table (combines shifts and sessions)
CREATE TABLE IF NOT EXISTS shift_sessions (
    id SERIAL PRIMARY KEY,
    shift_session_id UUID DEFAULT uuid_generate_v4() UNIQUE,
    shift_name VARCHAR(20) NOT NULL CHECK (
        shift_name IN ('`', '``', 'RBS1', 'RBS2')
    ),
    clerk_initials VARCHAR(10) NOT NULL,
    session_date DATE NOT NULL,
    start_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    end_time TIMESTAMP,
    status VARCHAR(10) NOT NULL DEFAULT 'OPEN' CHECK (status IN ('OPEN', 'CLOSED')),
    closed_by VARCHAR(50),
    is_active BOOLEAN DEFAULT true,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(shift_name, session_date, clerk_initials)
);

-- 4. Bill number sequence table (unchanged)
CREATE TABLE IF NOT EXISTS bill_sequences (
    bill_date DATE PRIMARY KEY,
    last_number INTEGER DEFAULT 0
);

-- 5. Bills table - UPDATED (removed shift_id, session_id, modified_from_bill_id, updated_at)
CREATE TABLE IF NOT EXISTS bills (
    id SERIAL PRIMARY KEY,
    bill_number INTEGER NOT NULL,
    bill_date DATE NOT NULL,
    table_no VARCHAR(20),
    party_no VARCHAR(20) DEFAULT '1',
    section VARCHAR(10) DEFAULT 'G',
    track VARCHAR(20),
    clerk_initials VARCHAR(10),
    subtotal DECIMAL(10,2) DEFAULT 0,
    sgst DECIMAL(10,2) DEFAULT 0,
    cgst DECIMAL(10,2) DEFAULT 0,
    tax_amount DECIMAL(10,2) DEFAULT 0,
    grand_total DECIMAL(10,2) DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(bill_number, bill_date)
);

-- 6. Bill items table (unchanged)
CREATE TABLE IF NOT EXISTS bill_items (
    id SERIAL PRIMARY KEY,
    bill_id INTEGER REFERENCES bills(id) ON DELETE CASCADE,
    item_code VARCHAR(20),
    item_name VARCHAR(255),
    quantity INTEGER DEFAULT 1,
    unit_price DECIMAL(10,2) DEFAULT 0,
    line_total DECIMAL(10,2) DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 7. Audit log table for tracking all actions (removed session_id reference)
CREATE TABLE IF NOT EXISTS audit_log (
    id SERIAL PRIMARY KEY,
    event_id UUID DEFAULT uuid_generate_v4(),
    timestamp_utc TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    performed_by_user_id VARCHAR(50),
    performed_by_user_name VARCHAR(100),
    user_role VARCHAR(20),
    action_type VARCHAR(50),
    resource_type VARCHAR(50),
    resource_id VARCHAR(50),
    shift_session_id UUID REFERENCES shift_sessions(shift_session_id),
    ip_address INET,
    payload JSONB,
    correlation_id UUID
);

-- Create indexes for performance
CREATE INDEX IF NOT EXISTS idx_bills_date_number ON bills(bill_date, bill_number);
CREATE INDEX IF NOT EXISTS idx_bills_clerk ON bills(clerk_initials, bill_date);
CREATE INDEX IF NOT EXISTS idx_items_codes ON items(alpha_code, numeric_code);
CREATE INDEX IF NOT EXISTS idx_shift_sessions_active ON shift_sessions(is_active, session_date);
CREATE INDEX IF NOT EXISTS idx_shift_sessions_date_status ON shift_sessions(session_date, status);
CREATE INDEX IF NOT EXISTS idx_shift_sessions_name_date ON shift_sessions(shift_name, session_date);

-- Helper function to get current open shift session for a given shift type and date
CREATE OR REPLACE FUNCTION get_current_shift_session_id(
    shift_type VARCHAR DEFAULT NULL, 
    target_date DATE DEFAULT NULL,
    clerk_initials_param VARCHAR DEFAULT NULL
) RETURNS UUID AS $function_body$
DECLARE
    shift_session_id UUID;
    func_date DATE;
    func_time TIME;
BEGIN
    -- Use provided date or current date
    func_date := COALESCE(target_date, CURRENT_DATE);
    func_time := CURRENT_TIME;
    
    -- If shift_type is not specified, determine based on current time
    IF shift_type IS NULL THEN
        IF func_time BETWEEN TIME '06:00:00' AND TIME '11:59:59' THEN
            shift_type := '`';
        ELSIF func_time BETWEEN TIME '12:00:00' AND TIME '17:59:59' THEN
            shift_type := '``';
        ELSIF func_time BETWEEN TIME '18:00:00' AND TIME '21:59:59' THEN
            shift_type := 'RBS1';
        ELSE
            shift_type := 'RBS2';
        END IF;
    END IF;
    
    -- Get the shift session ID for the specified type and date where status is OPEN
    SELECT ss.shift_session_id INTO shift_session_id
    FROM shift_sessions ss
    WHERE ss.shift_name = shift_type 
      AND ss.session_date = func_date 
      AND ss.status = 'OPEN'
      AND (clerk_initials_param IS NULL OR ss.clerk_initials = clerk_initials_param)
    ORDER BY ss.start_time DESC
    LIMIT 1;
    
    -- If no open shift session found, try to find any shift session for that day/type
    IF shift_session_id IS NULL THEN
        SELECT ss.shift_session_id INTO shift_session_id
        FROM shift_sessions ss
        WHERE ss.shift_name = shift_type 
          AND ss.session_date = func_date
          AND (clerk_initials_param IS NULL OR ss.clerk_initials = clerk_initials_param)
        ORDER BY ss.start_time DESC
        LIMIT 1;
    END IF;
    
    RETURN shift_session_id;
END;
$function_body$ LANGUAGE plpgsql;

BEGIN
    INSERT INTO shift_sessions (shift_name, clerk_initials, session_date, start_time, status) 
    SELECT '`', 'SYS', CURRENT_DATE, CURRENT_TIMESTAMP, 'OPEN'

    INSERT INTO shift_sessions (shift_name, clerk_initials, session_date, start_time, status) 
    SELECT '``', 'SYS', CURRENT_DATE, CURRENT_TIMESTAMP, 'OPEN'
    WHERE NOT EXISTS (SELECT 1 FROM shift_sessions WHERE shift_name = '``' AND session_date = CURRENT_DATE);

    SELECT 'RBS1', 'SYS', CURRENT_DATE, CURRENT_TIMESTAMP, 'OPEN'

    SELECT 'RBS2', 'SYS', CURRENT_DATE, CURRENT_TIMESTAMP, 'OPEN'
    WHERE NOT EXISTS (SELECT 1 FROM shift_sessions WHERE shift_name = 'RBS2' AND session_date = CURRENT_DATE);
END;
