-- Helper function to get current open shift for a given shift type and date
-- This function is corrected for PostgreSQL 17.6 compatibility
CREATE OR REPLACE FUNCTION get_current_shift_id(
    shift_type VARCHAR DEFAULT NULL, 
    target_date DATE DEFAULT NULL
) RETURNS UUID AS $function_body$
DECLARE
    shift_id UUID;
    func_date DATE;
    func_time TIME;
BEGIN
    -- Use provided date or current date
    func_date := COALESCE(target_date, CURRENT_DATE);
    func_time := CURRENT_TIME;
    
    -- If shift_type is not specified, determine based on current time
    IF shift_type IS NULL THEN
        IF func_time BETWEEN TIME '06:00:00' AND TIME '11:59:59' THEN
            shift_type := '`';
        ELSIF func_time BETWEEN TIME '12:00:00' AND TIME '17:59:59' THEN
            shift_type := '``';
        ELSIF func_time BETWEEN TIME '18:00:00' AND TIME '21:59:59' THEN
            shift_type := 'RBS1';
        ELSE
            shift_type := 'RBS2';
        END IF;
    END IF;
    
    -- Get the shift ID for the specified type and date where status is OPEN
    SELECT s.shift_id INTO shift_id
    FROM shifts s
    WHERE s.shift_name = shift_type 
      AND s.date = func_date 
      AND s.status = 'OPEN'
    LIMIT 1;
    
    -- If no open shift found, try to find any shift for that day/type
    IF shift_id IS NULL THEN
        SELECT s.shift_id INTO shift_id
        FROM shifts s
        WHERE s.shift_name = shift_type 
          AND s.date = func_date
        LIMIT 1;
    END IF;
    
    RETURN shift_id;
END;
$function_body$ LANGUAGE plpgsql;
